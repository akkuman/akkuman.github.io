    switch反汇编(C语言)

    Written by Akkuman with ♥ on in 逆向

    在分支较多的时候，switch的效率比if高，在反汇编中我们即可看到效率高的原因

    0x01分支结构不超过3个

    #include <stdio.h>

void main()
{
	int x = 5;
	switch(x)
	{
	case 5:
		printf("%d\n",x);
		break;
	case 6:
		printf("%d\n",x);
		break;
	case 7:
		printf("%d\n",x);
		break;
	default:
		break;
	}

	return;
}

    0x03分支跃度大难以构成跳转表的分支结构

    #include <stdio.h>

void main()
{
	int x = 5;
	switch(x)
	{
	case 5:
		printf("%d\n",x);
		break;
	case 6:
		printf("%d\n",x);
		break;
	case 7:
		printf("%d\n",x);
		break;
	case 8:
		printf("%d\n",x);
		break;
	case 100:
		printf("%d\n",x);
		break;
	default:
		printf("%d\n",x);
		break;
	}

	return;
}
