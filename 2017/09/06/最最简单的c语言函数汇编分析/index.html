push      ebp
mov       ebp,esp
sub       esp,40h
//提升栈，为函数腾出空间，为ebp寻址做准备
push      ebx
push      esi
push      edi
//寄存器压栈，保存现场
lea       edi,[ebp-40h]
//将ebp-40h（esp）的具体内存地址存到edi
mov       ecx,10h
//10（十六进制）存入计数寄存器
mov       eax,0xCCCCCCCC
//初始化eax
rep stos  dword ptr [edi]
//用eax中的值初始化到es:[edi]指向的地址，长度为dword，循环执行次数为eax中的值（恰好ebp->esp全部被初始化）
xor       eax,eax
//eax清零
pop       edi
pop       esi
pop       ebx
mov       esp,ebp
pop       ebp
ret
//寄存器出栈，恢复现场，堆栈平衡并返回
